118  llvm::DenseMap<BasicBlock *, BBInOutBits *> previousInOutMap;

122  initializeBbBitMaps(m_func, previousInOutMap);

124  if (m_dir == FORWARD) {

142 template <typename D>
143 void DataflowFramework<D>::initializeBbBitMaps(
144  Function &F, llvm::DenseMap<BasicBlock *, BBInOutBits *> &map) {
145  BasicBlock &entry = F.getEntryBlock();
146  BasicBlock &exit = F.getBasicBlockList().back();
147  // Usually we prefer not to use pointers because owners
148  // of the memory are not very obvious. In this case, we
149  // have to use pointers because of the DenseMap. This is
150  // ok as long as we don't delete entries from the map.
151  // As for the pair itself, the DenseMap will copy the
152  // pair in, so even if the pair goes out of scope, there
153  // is no issue.
154  if (m_dir == FORWARD) {
155  for (BasicBlock &BB : F) {
156  BBInOutBits *p_inOut =
157  new BBInOutBits(ZEROS, m_meetOp.getTopElem());
158  std::pair<BasicBlock *, BBInOutBits *> p_pair;
159  p_pair.first = &BB;
160  p_pair.second = p_inOut;
161  map.insert(p_pair);
162  }
163  // Explicitly clear or set the entry block's IN, while the
164  // algorithm requires we clear/set an entry block's OUT, there
165  // is no empty entry block, in the conventional sense in LLVM.
166  // This is the first block, which is effectively the first block
167  // after entry:
168  if (m_boundary == UNIVERSAL) {
169  map.find(&entry)->second->m_IN.set();
170  } else if (m_boundary == EMPTY) {
171  map.find(&entry)->second->m_IN.reset();
172  }
173  } else {
174  for (BasicBlock &BB : F) {
175  BBInOutBits *p_inOut =
176  new BBInOutBits(m_meetOp.getTopElem(), ZEROS);
177  std::pair<BasicBlock *, BBInOutBits *> p_pair;
178  p_pair.first = &BB;
179  p_pair.second = p_inOut;
180  map.insert(p_pair);
181  }
182  // Explicitly set the exit block's OUT, again, this doesn't
183  // cover multiple exit points and assumes a single return in the
184  // function, we can add a sentinel "exit" node later on if
185  // required for our analysis. Again, while the algorithm
186  // requires we set the "exit" block's IN, we don't have an empty
187  // "exit" block, the "exit" block currently is just the last
188  // block of the function, containing the return instruction,
189  // thus, we set the OUT of the "exit" block:
190  if (m_boundary == UNIVERSAL) {
191  map.find(&exit)->second->m_OUT.set();
192  } else if (m_boundary == EMPTY) {
193  map.find(&exit)->second->m_OUT.reset();
194  }
195  }
196 }

199 void DataflowFramework<D>::deepCopyDenseMaps(
200  llvm::DenseMap<BasicBlock *, BBInOutBits *> &currentMap,
201  llvm::DenseMap<BasicBlock *, BBInOutBits *> &previousMap) {
202  for (auto it = currentMap.begin(); it != currentMap.end(); ++it) {
203  // Get previous mapping
204  auto previous = previousMap.find(it->first);
205  if (previous != previousMap.end()) {
206  previous->second->m_OUT = it->second->m_OUT;
207  previous->second->m_IN = it->second->m_IN;
208  } else {
209  // Error, BB entry in current not found in previous,
210  // should be impossible
211  llvm_unreachable(
212  "BB Entry in Current not found in Previous");
213  }
214  }

307 template <typename D>
308 void DataflowFramework<D>::doForwardTraversal(
309  llvm::DenseMap<BasicBlock *, BBInOutBits *> &currentInOutMap,
310  llvm::DenseMap<BasicBlock *, BBInOutBits *> &previousInOutMap) {
311  // For now this will suffice, but we need to be wary of multiple return
312  // statements, if that is present in the program. In LLVM there can be
313  // multiple terminator instructions that are returns in a basic block.
314  // There is no 'EXIT' block for the procedure. This means our ipo_begin
315  // instantiation might not be 100% accurate. This also applies for
316  // post_order.
317  BasicBlock *BB;
318  std::bitset<MAX_BITS_SIZE> meet_res, BB_killset, BB_genset;
319  int iteration_no = 1;
320  do {
321  outs() << "ITERATION " << iteration_no << "\n";
322  outs() << "++++++++++++++++++++++++++++++++++++++++++++++++++++"
323  "++++++++++++++++++++++++++++++++"
324  << "\n";
325  // Make values of previous == current
326  deepCopyDenseMaps(currentInOutMap, previousInOutMap);
327 
328  for (ipo_iterator<BasicBlock *> I =
329  ipo_begin(&m_func.getBasicBlockList().back());
330  I != ipo_end(&m_func.getEntryBlock()); ++I) {
331 
332  if (BB = dyn_cast<BasicBlock>(*I))
333  outs() << *BB << "\n";
334  // Pre-define for convenience so we don't have to keep
335  // looking it up
336  BBInOutBits *currentInOutBits = currentInOutMap[BB];
337  // Actually, normally meet of all predecessors vanilla
338  // would work, but we need to initialize one of them to
339  // our m_IN. Recall that we don't actually have an empty
340  // ENTRY block. Our ENTRY block in llvm is actually
341  // conceptually the block after the empty ENTRY block.
342  // That's why we initialized the IN of this block to the
343  // Universal or Empty set. Making the initial met_res
344  // equal to the IN of the current block solves this for
345  // entry block. Else, if we're any other block, we take
346  // any of the predecessors and make the meet_res equal
347  // to that.
348  if (BB == &m_func.getEntryBlock()) {
349  meet_res = currentInOutBits->m_IN;
350  } else {
351  auto pred = pred_begin(BB);
352  meet_res = currentInOutMap[*pred]->m_OUT;
353  }
354  // MEET OF ALL PREDECESSORS
355  for (BasicBlock *Pred : predecessors(BB)) {
356  BBInOutBits *ip1 = currentInOutMap[Pred];
357  meet_res = m_meetOp.meet(ip1->m_OUT, meet_res);
358  }
359 
360  outs() << "Current IN bits: "
361  << meet_res.to_string().substr(
362  MAX_BITS_SIZE - MAX_PRINT_SIZE,
363  MAX_BITS_SIZE)
364  << "\n";
365 
366  // Create genset and killset
367  BB_genset = m_KG.genEval(BB, meet_res, m_domainSet);
368  outs() << "Current GEN Set: "
369  << BB_genset.to_string().substr(
370  MAX_BITS_SIZE - MAX_PRINT_SIZE,
371  MAX_BITS_SIZE)
372  << "\n";
373 
374  BB_killset = m_KG.killEval(BB, meet_res, m_domainSet);
375  outs() << "Current KILL Set: "
376  << BB_killset.to_string().substr(
377  MAX_BITS_SIZE - MAX_PRINT_SIZE,
378  MAX_BITS_SIZE)
379  << "\n";
380 
381  // Run transfer function on our sets, store the bits:
382  currentInOutBits->m_OUT =
383  m_transferFunc.run(meet_res, BB_genset, BB_killset);
384  outs() << "Current OUT bits: "
385  << currentInOutBits->m_OUT.to_string().substr(
386  MAX_BITS_SIZE - MAX_PRINT_SIZE,
387  MAX_BITS_SIZE)
388  << "\n";
389  outs() << "======================================"
390  "======================================"
391  << "\n";
392  }
393  ++iteration_no;
394  } while (hasOutChanged(currentInOutMap, previousInOutMap));
395 }
